    f(x) &= \lim_{n \to \infty} \left| \frac{a_{n+1}}{a_n} \right| \\
         &= \lim_{n \to \infty} \left| \frac{\left| x(n+1) !(x-1)^{1/} + 1 \right|}{(n+1) \frac{n}{x n !(x-1)^n}} \right| \\
         &= \lim_{n \to \infty} \left| \frac{(n+1)^{n !(x-1)^n(x-1)^m n}}{(n+1)^{n+(x)}} \right| \\
         &= |x - 1| \lim_{n \to \infty} n > 1 \text{ always diverges}
    \end{aligned}
