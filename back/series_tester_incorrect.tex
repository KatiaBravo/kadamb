\begin{aligned}
    & \lim_{n=1}^{\infty} (-1)^n \frac{4n!(x-1)^n}{n} = a_n = \frac{4n!(x-1)^n}{n} \\
    & \lim_{n \to \infty} \left| \frac{a_{n+1}}{a_n} \right| = \lim_{n \to \infty} \left| \frac{4(n+1)!(x-1)^{n+1}}{n+1} \cdot \frac{n}{4n!(x-1)^n} \right| \\
    & = \lim_{n \to \infty} \left| \frac{(n+1) n!(x-1)(x-1)^n n}{(n+1) 4n!(x-1)^n} \right| \\
    & = \lim_{n \to \infty} |(x-1) n| = |x-1| \lim_{n \to \infty} n \\
    & |x-1| > 1 \text{, therefore it diverges.}
    \end{aligned}
